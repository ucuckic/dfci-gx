{

	local flag = BMvTbl.GetPP( def_PP_Krt_AddSkill );
	if( flag&(1<<_skillnum) )
	{
		// _dp("\n ÉtÉâÉOÇΩÇ¡ÇƒÇÈÇÃÇ≈É_ÉÅ");
		return 0;
	}
	// ïKéEÇbÇ™ÇgÇhÇséûÇÃéûÇÃÇ›èoÇπÇÈ

	return 1;
